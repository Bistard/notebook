# 简单数学-Simple Math

```js
let num;
// addition
num = 5 + 10;
num++;

// subtraction
num = num - 5;
num -= 5;
num--;

// product
num = num * 5;
num *= 5;

// division
num = num / 2; /* error num = num / 0; */
num /= 2;

// decimals
let decimal;
decimal = 5.0;
decimal = 3.1415926535;
decimal = 2.0 * 5.0;
decimal = 0.1 * 1.1;

// remiander
let remainder;
remainder = 11 % 5; // 1
```
